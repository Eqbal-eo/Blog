blog-project/
├── controllers/
│   └── authController.js         # التحكم في المصادقة (تسجيل الدخول، الخروج)
├── db/
│   └── db.js                     # ملف الاتصال بقاعدة البيانات
├── node_modules/                 # مجلد يحتوي على التبعيات المثبتة عبر npm
├── public/
│   └── css/
│       ├── login.css             # CSS لصفحة تسجيل الدخول
│       ├── home.css              # CSS للصفحة الرئيسية
│       ├── user-blog.css         # CSS لصفحة مدونة المستخدم
│       ├── dashboard.css         # CSS لصفحة لوحة التحكم
│       ├── editor.css            # CSS لتخصيص المحرر في نظام التحكم
│       └── footer.css            # CSS خاص بمواقع العرض
├── routes/
│   ├── authRoutes.js             # مسارات المصادقة (تسجيل الدخول والخروج)
│   ├── postRoutes.js             # مسارات التدوينات (إنشاء، تعديل، حذف، عرض)
│   └── mainRoutes.js             # مسارات المستخدم (عرض المدونة الخاصة بالمستخدم)
├── uploads/                      # مجلد لتخزين الملفات المرفوعة (مثل الصور)
├── views/
│   ├── create-post.ejs           # صفحة إنشاء تدوينة جديدة
│   ├── dashboard.ejs             # لوحة التحكم للمستخدم (إدارة التدوينات)
│   ├── home.ejs                  # الصفحة الرئيسية (عرض التدوينات لجميع المستخدمين)
│   ├── login.ejs                 # صفحة تسجيل الدخول
│   ├── register.ejs              # صفحة تسجيل حساب جديد
│   ├── user-blog.ejs             # صفحة عرض مدونة المستخدم (التدوينات + النبذة)
│   ├── settings.ejs              # صفحة إعدادات الحساب (النبذة والتواصل)
│   ├── select-delete.ejs         # صفحة اختيار التدوينة لحذفها
│   ├── select-edit.ejs           # صفحة اختيار التدوينة لتعديلها
│   ├── contact.ejs               # صفحة الاتصال
│   ├── about.ejs                 # صفحة عني
│   └── post.ejs                  # صفحة عرض التدوينة بعد الضغط على "قراءة المزيد"
├── app.js                        # نقطة البداية للتطبيق، يحتوي على إعدادات Express
├── package.json                  # ملف إعدادات npm مع التبعيات
└── package-lock.json             # قفل الإصدارات لجميع التبعيات
